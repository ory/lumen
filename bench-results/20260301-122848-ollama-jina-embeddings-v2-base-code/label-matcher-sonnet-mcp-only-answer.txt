```go
func NewMatcher(t MatchType, n, v string) (*Matcher, error) {
    m := &Matcher{
        Type:  t,
        Name:  n,
        Value: v,
    }
    if t == MatchRegexp || t == MatchNotRegexp {
        re, err := NewFastRegexMatcher(v)
        if err != nil {
            return nil, err
        }
        m.re = re
    }
    return m, nil
}
```

For `MatchRegexp` / `MatchNotRegexp`, the value string is compiled into a `FastRegexMatcher` at construction time. Returns an error if the regex is invalid.
